CREATE TABLE parents(
	id serial PRIMARY KEY,
	firstName varchar(50),
	lastName varchar(50),
	dateBirth DATE
);

CREATE TABLE children(
	children_id serial PRIMARY KEY,
	parent_id integer REFERENCES parents(id),
	firstName varchar(50),
	lastName varchar(50),
	dateBirth DATE 
);
